function mobileOnlySlider($slidername, $dots, $arrows, $breakpoint) {
	var slider = $($slidername);
	var settings = {
		mobileFirst: true, 
  infinite: false,
		arrows: $arrows,
  slidesToShow: 1.25,
		responsive: [
			{
				breakpoint: $breakpoint,
				settings: "unslick"
			}
		]
	};

	slider.slick(settings);

	$(window).on("resize", function () {
		if ($(window).width() > $breakpoint) {
			return;
		}
		if (!slider.hasClass("slick-initialized")) {
			return slider.slick(settings);
		}
	});
} // Mobile Only Slider

document.querySelector('#wf-form-Multi-Class-Registration-Form').addEventListener('submit', async (event) => {
  event.preventDefault(); // Prevent the default form submission

  // Get the form element
  const form = event.target;

  // Create a FormData object from the form
  const formData = new FormData(form);

  // Convert FormData to a plain object
  const data = Object.fromEntries(formData.entries());

  // Default price per seat (Ensure this value is numeric)
  const pricePerSeat = parseFloat('{{wf {&quot;path&quot;:&quot;price-member&quot;,&quot;type&quot;:&quot;PlainText&quot;\} }}') || 
                      parseFloat('{{wf {&quot;path&quot;:&quot;price-non-member&quot;,&quot;type&quot;:&quot;PlainText&quot;\} }}');

  if (isNaN(pricePerSeat)) {
    alert('Error: Invalid price per seat. Please contact support.');
    return;
  }

  // Add pricePerSeat to the data object for backend validation
  data.pricePerSeat = pricePerSeat;

  // Add priceId to the data object
  const priceId = '{{wf {&quot;path&quot;:&quot;member-price-id&quot;,&quot;type&quot;:&quot;PlainText&quot;\} }}' || 
                  '{{wf {&quot;path&quot;:&quot;non-member-price-id&quot;,&quot;type&quot;:&quot;PlainText&quot;\} }}';

  if (!priceId) {
    alert('Error: Price ID is missing. Please contact support.');
    return;
  }

  data.priceId = priceId; // Add priceId to the data object

  // Calculate the seat count based on filled participant fields
  let seatCount = 0;
  for (let i = 1; i <= 10; i++) {
    const name = formData.get(`P${i}-Name`);
    if (name) seatCount++;
  }

  data.numberOfSeats = seatCount; // Add the calculated seat count to the data object

  try {
    // Log the form data to the console for debugging
    console.log('Form data submitted:', data);

    // Send the form data to your backend
    const response = await fetch('https://class-module-f55u.onrender.com/submit-class', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(data), // Convert the data to JSON
    });

    if (response.ok) {
      const result = await response.json();
      console.log('Success:', result);

      // Redirect to the Stripe Checkout URL
      if (result.checkoutUrl) {
        window.location.href = result.checkoutUrl; // Redirect user to Stripe Checkout
      } else {
        alert('Error: Checkout URL not provided by the server. Please try again.');
      }
    } else {
      const errorData = await response.json();
      console.error('Server responded with an error:', response.status, errorData);
      alert('Failed to process the form. Please try again.');
    }
  } catch (error) {
    console.error('Error:', error);
    alert('An unexpected error occurred. Please try again.');
  }
});

</script>

document.addEventListener('DOMContentLoaded', () => {
  // Helper function to check if an element is visible
  const isElementVisible = (element) => {
    return element &&
      window.getComputedStyle(element).display !== 'none' &&
      !element.classList.contains('w-condition-invisible');
  };

  // Select all elements with the 'cancel-form' class
  const cancelForms = document.querySelectorAll('.cancel-form');

  cancelForms.forEach((cancelForm) => {
    // Find the nearest parent containing the cancel form
    const parent = cancelForm.closest('.purchase-details_button-wrapper');
    if (!parent) return;

    // Query the cancelled, refunded, and cancelled_without divs
    const cancelledDiv = parent.querySelector('.cancelled');
    const refundedDiv = parent.querySelector('.refunded');
    const cancelledWithoutDiv = parent.querySelector('.cancelled_without');

    // Skip if none of the relevant elements are found
    if (!cancelledDiv && !refundedDiv && !cancelledWithoutDiv) {
      console.warn('Relevant elements not found for:', cancelForm);
      return;
    }

    // Check visibility of the elements
    const isCancelledVisible = isElementVisible(cancelledDiv);
    const isRefundedVisible = isElementVisible(refundedDiv);
    const isCancelledWithoutVisible = isElementVisible(cancelledWithoutDiv);

    // Update the cancel form visibility based on the conditions
    if (isCancelledVisible || isRefundedVisible || isCancelledWithoutVisible) {
      // Hide the cancel form if any of the conditions are met
      cancelForm.classList.remove('block');
      cancelForm.classList.add('hide');
    } else {
      // Show the cancel form otherwise
      cancelForm.classList.remove('hide');
      cancelForm.classList.add('block');
    }

    // Debugging logs for visibility status
    const debug = false; // Set to true for debugging
    if (debug) {
      console.log('Cancel Form:', cancelForm);
      console.log('Cancelled Visible:', isCancelledVisible);
      console.log('Refunded Visible:', isRefundedVisible);
      console.log('Cancelled Without Visible:', isCancelledWithoutVisible);
    }
  });
});

</script>
