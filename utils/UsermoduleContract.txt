
///////////////////////////////////////////////CONTRACT SUBSCRIPTION AUTOPAY////////////////////////////////////////////////////////

  document.addEventListener("DOMContentLoaded", function () {
  const checkbox = document.getElementById("checkbox-builder");
  const button = document.getElementById("autopaybtn");

  if (checkbox && button) {
    button.addEventListener("click", function () {
      const isChecked = checkbox.checked;

      // Get the member data from localStorage
      const memDataRaw = localStorage.getItem('_ms-mem');
      let memberId = null;

      try {
        const memData = JSON.parse(memDataRaw);
        memberId = memData?.id || null;
        
        if (!memberId) {
          console.error("No member ID found in localStorage");
          alert("Error: Member ID not found. Please log in again.");
          return;
        }
      } catch (err) {
        console.error("Error parsing _ms-mem from localStorage:", err);
        alert("Error: Could not retrieve member information. Please log in again.");
        return;
      }

      const payload = {
        autopayDisabled: isChecked,
        memberId: memberId,
      };

      // Disable button while processing
      button.disabled = true;
      button.textContent = "Updating...";

      // Log the data being sent
      console.log("Sending data to server:", payload);

      fetch("https://biaw-server-hbwi.onrender.com/api/product-autopay/update-status", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify(payload),
      })
      .then(response => {
        if (!response.ok) {
          throw new Error(`HTTP error! status: ${response.status}`);
        }
        return response.json();
      })
      .then(data => {
        console.log("Server response:", data);
        // Show success message
        alert(isChecked ? "Autopay has been disabled" : "Autopay has been enabled");
      })
      .catch(error => {
        console.error("Error sending checkbox state:", error);
        alert("Error updating autopay status. Please try again.");
        // Revert checkbox state on error
        checkbox.checked = !isChecked;
      })
      .finally(() => {
        // Re-enable button
        button.disabled = false;
        button.textContent = "Update";
      });
    });
  } else {
    console.error("Checkbox or button not found.");
  }
});
